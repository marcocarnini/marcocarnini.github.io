The current model

Starting from the previous post and refactoring, the current model is:

import numpy as np
import pandas as pd
from sklearn.ensemble import RandomForestClassifier
from sklearn.model_selection import cross_val_score
from sklearn.model_selection import KFold
from sklearn import preprocessing
from category_encoders import CatBoostEncoder


def read_data() -> tuple:
    return (
        pd.read_csv("../input/4910797b-ee55-40a7-8668-10efd5c1b960.csv"),
        pd.read_csv("../input/702ddfc5-68cd-4d1d-a0de-f5f566f76d91.csv"),
        pd.read_csv("../input/0bf8bc6e-30d0-4c50-956a-603fc693d966.csv"),
    )


def clean_data(train: pd.DataFrame, test: pd.DataFrame) -> tuple:
    train.drop(["recorded_by"], axis=1, inplace=True)
    test.drop(["recorded_by"], axis=1, inplace=True)

    train["source"] = "train"
    test["source"] = "test"
    temp = pd.concat(
        (
            train[
                [
                    "source",
                    "id",
                    "latitude",
                    "longitude",
                    "region_code",
                    "district_code",
                ]
            ],
            test[
                [
                    "source",
                    "id",
                    "latitude",
                    "longitude",
                    "region_code",
                    "district_code",
                ]
            ],
        )
    )

    region_imputation = (
        temp.loc[train["longitude"] >= 29, ["latitude", "longitude", "region_code"]]
        .groupby("region_code")
        .agg(["mean", "median"])
    )
    region_imputation.columns = [
        "r-latitude-mean",
        "r-latitude-median",
        "r-longitude-mean",
        "r-longitude-median",
    ]

    district_imputation = (
        temp.loc[train["longitude"] >= 29, ["latitude", "longitude", "district_code"]]
        .groupby("district_code")
        .agg(["mean", "median"])
    )
    district_imputation.columns = [
        "d-latitude-mean",
        "d-latitude-median",
        "d-longitude-mean",
        "d-longitude-median",
    ]

    missing = (
        temp.loc[temp["longitude"] < 29, :]
        .merge(region_imputation, how="left", on="region_code")
        .merge(district_imputation, how="left", on="district_code")
    )
    missing.drop(
        ["latitude", "longitude", "region_code", "district_code"], axis=1, inplace=True
    )

    nonmissing = temp.loc[temp["longitude"] >= 29, :].copy()
    nonmissing["r-latitude-mean"] = nonmissing["latitude"]
    nonmissing["r-latitude-median"] = nonmissing["latitude"]
    nonmissing["r-longitude-mean"] = nonmissing["longitude"]
    nonmissing["r-longitude-median"] = nonmissing["longitude"]
    nonmissing["d-latitude-mean"] = nonmissing["latitude"]
    nonmissing["d-latitude-median"] = nonmissing["latitude"]
    nonmissing["d-longitude-mean"] = nonmissing["longitude"]
    nonmissing["d-longitude-median"] = nonmissing["longitude"]
    nonmissing.drop(
        ["latitude", "longitude", "region_code", "district_code"], axis=1, inplace=True
    )

    imputed = pd.concat((missing, nonmissing))
    train_impute = imputed.loc[imputed["source"] == "train",].drop(["source"], axis=1)
    test_impute = imputed.loc[imputed["source"] == "test",].drop(["source"], axis=1)
    train = train.merge(train_impute, how="left", on="id")
    test = test.merge(test_impute, how="left", on="id")

    X = train.select_dtypes(include=["int64", "float64"]).loc[
        train["construction_year"] > 0, :
    ]
    Y = X.construction_year.ravel()
    X.drop(["construction_year"], axis=1, inplace=True)
    model = RandomForestClassifier(random_state=2020).fit(X, Y)

    train["Construction_imputed"] = model.predict(
        train.select_dtypes(include=["int64", "float64"]).drop(
            ["construction_year"], axis=1
        )
    )
    test["Construction_imputed"] = model.predict(
        test.select_dtypes(include=["int64", "float64"]).drop(
            ["construction_year"], axis=1
        )
    )
    train["Construction_year_missing"] = (train["construction_year"] == 0).astype(int)
    test["Construction_year_missing"] = (test["construction_year"] == 0).astype(int)

    return train, test, model


def train_model(train: pd.DataFrame, labels: pd.DataFrame) -> list:
    kf = KFold(n_splits=10, shuffle=True, random_state=2020)
    train["population_group"] = pd.Series(
        np.where(train["population"] > 1, 2, train["population"])
    ).map({0: "0", 1: "1", 2: ">1"})

    X = train[[
    "id",
    "amount_tsh",
    "gps_height",
    "longitude",
    "latitude",
    "num_private",
    "region_code",
    "district_code",
    "population",
    "construction_year",
    "r-latitude-mean",
    "r-latitude-median",
    "r-longitude-mean",
    "r-longitude-median",
    "d-latitude-mean",
    "d-latitude-median",
    "d-longitude-mean",
    "d-longitude-median",
    "Construction_imputed",
    "Construction_year_missing",
    "population_group"
    ]]


    le = preprocessing.LabelEncoder()
    le.fit(['functional', 'functional needs repair', 'non functional'])
    Y = le.transform(labels.status_group.ravel())

    acc = []
    for train_index, test_index in kf.split(X):
        X_train, X_test = X.loc[train_index, :], X.loc[test_index, :]
        Y_train, Y_test = Y[train_index], Y[test_index]
    
        encoder = CatBoostEncoder(cols=["population_group"])
        encoder.fit(X_train, Y_train)
        X_train = encoder.transform(X_train)
        X_test = encoder.transform(X_test)
        model = RandomForestClassifier(random_state=2020).fit(X_train, Y_train)
        acc += [np.mean(Y_test == model.predict(X_test))]
    
    return acc


train, test, labels = read_data()
train, test, model = clean_data(train, test)
rf_scores = train_model(train, labels)
print(np.mean(rf_scores), "+/-", np.std(rf_scores))

Treating region_code and district_code as categorical

So far, the district and region code were treated as numbers, as some informations can be extracted (for example, in the Italian region codes). What I think is a more reasonable approach is to treat them as categories, and encoding them (CatBoost encoding).

def train_model(train: pd.DataFrame, labels: pd.DataFrame) -> list:
    kf = KFold(n_splits=10, shuffle=True, random_state=2020)
    train["population_group"] = pd.Series(
        np.where(train["population"] > 1, 2, train["population"])
    ).map({0: "0", 1: "1", 2: ">1"})

    X = train[
        [
            "id",
            "amount_tsh",
            "gps_height",
            "longitude",
            "latitude",
            "num_private",
            "region_code",
            "district_code",
            "population",
            "construction_year",
            "r-latitude-mean",
            "r-latitude-median",
            "r-longitude-mean",
            "r-longitude-median",
            "d-latitude-mean",
            "d-latitude-median",
            "d-longitude-mean",
            "d-longitude-median",
            "Construction_imputed",
            "Construction_year_missing",
            "population_group",
        ]
    ].copy()
    X["district_code_cat"] = X["district_code"].apply(str)
    X["region_code_cat"] = X["region_code"].apply(str)

    le = preprocessing.LabelEncoder()
    le.fit(["functional", "functional needs repair", "non functional"])
    Y = le.transform(labels.status_group.ravel())

    acc = []
    for train_index, test_index in kf.split(X):
        X_train, X_test = X.loc[train_index, :], X.loc[test_index, :]
        Y_train, Y_test = Y[train_index], Y[test_index]

        encoder = CatBoostEncoder(
            cols=["population_group", "district_code_cat", "region_code_cat"]
        )
        encoder.fit(X_train, Y_train)
        X_train = encoder.transform(X_train)
        X_test = encoder.transform(X_test)
        model = RandomForestClassifier(random_state=2020).fit(X_train, Y_train)
        acc += [np.mean(Y_test == model.predict(X_test))]

    return acc


rf_scores = train_model(train, labels)
print(np.mean(rf_scores), "+/-", np.std(rf_scores))
